## helpers.js (pure)

```js
export const debounce = (fn, ms = 300) => {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), ms);
  };
};

export const excerpt = (text, max = 100) =>
  text.length > max ? `${text.slice(0, max).trimEnd()}…` : text;

export const byId = (list) =>
  list.reduce((map, item) => ({ ...map, [item.id]: item }), {});
```

## post-list.js

```js
export function createPostList(root) {
  let cards = [];

  const observer = new IntersectionObserver((entries) => {
    entries.forEach((entry) => {
      if (!entry.isIntersecting) return;
      entry.target.classList.add('card--visible');
      observer.unobserve(entry.target);
    });
  });

  const cardEl = (post) => {
    const article = document.createElement('article');
    article.className = 'card';
    article.innerHTML = `
      <h2 class="card__title">${post.title}</h2>
      <p class="card__author">by ${post.author}</p>
      <p class="card__excerpt">${post.excerpt}</p>`;
    article.addEventListener('click', () => {
      root.dispatchEvent(new CustomEvent('post:open', { detail: post, bubbles: true }));
    });
    observer.observe(article);
    return article;
  };

  return {
    render(posts) {
      cards = posts.map((post) => ({ post, el: cardEl(post) }));
      root.replaceChildren(...cards.map((c) => c.el));
    },
    filter(query) {
      cards.forEach(({ post, el }) => {
        const match = post.title.toLowerCase().includes(query)
          || post.author.toLowerCase().includes(query);
        el.hidden = !match;
      });
    },
    error(message) {
      root.textContent = message;
    },
  };
}
```

## post-dialog.js

```js
import { getComments } from './api';

export function createPostDialog(root) {
  const comments = root.querySelector('.dialog__comments');

  root.querySelector('.dialog__close')
    .addEventListener('click', () => root.close());

  document.addEventListener('post:open', async (event) => {
    const post = event.detail;
    root.querySelector('.dialog__title').textContent = post.title;
    root.querySelector('.dialog__author').textContent = `by ${post.author}`;
    root.querySelector('.dialog__body').textContent = post.body;
    comments.innerHTML = '<li class="dialog__comment">Loading…</li>';
    root.showModal();

    try {
      const list = await getComments(post.id);
      comments.replaceChildren(...list.map((c) => {
        const li = document.createElement('li');
        li.className = 'dialog__comment';
        li.textContent = `${c.name}: ${c.body}`;
        return li;
      }));
    } catch {
      comments.innerHTML = '<li class="dialog__comment">Could not load comments.</li>';
    }
  });
}
```

## blog-explorer.clientlibs.js (the entry — the `main.js` equivalent)

```js
import { register } from '@netcentric/component-loader';
import { getPosts, getUsers } from './api';
import { byId, excerpt } from './helpers';
import { createSearchBar } from './search-bar';
import { createPostList } from './post-list';
import { createPostDialog } from './post-dialog';

async function BlogExplorer(element) {
  const list = createPostList(element.querySelector('.blog-explorer__list'));
  createPostDialog(element.querySelector('.blog-explorer__dialog'));
  createSearchBar(element.querySelector('.blog-explorer__search'), {
    onSearch: (query) => list.filter(query),
  });

  try {
    const posts = await getPosts();          // sequential — no Promise.all needed
    const users = await getUsers();
    const usersById = byId(users);
    const count = Number(element.dataset.count) || posts.length;

    const shaped = posts.slice(0, count).map((post) => ({
      id: post.id,
      title: post.title,
      body: post.body,
      author: usersById[post.userId]?.name ?? 'Unknown',
      excerpt: excerpt(post.body),
    }));

    list.render(shaped);
  } catch {
    list.error('Could not load posts.');
  }
}

register({ BlogExplorer });
export default BlogExplorer;
```

```java
package biz.netcentric.academy.core.models.blogexplorer;

import org.apache.sling.api.resource.Resource;
import org.apache.sling.models.annotations.Default;
import org.apache.sling.models.annotations.DefaultInjectionStrategy;
import org.apache.sling.models.annotations.Model;
import org.apache.sling.models.annotations.injectorspecific.ValueMapValue;

@Model(adaptables = Resource.class, defaultInjectionStrategy = DefaultInjectionStrategy.OPTIONAL)
public class BlogExplorerModel {

    @ValueMapValue @Default(values = "Blog Explorer")
    private String heading;

    @ValueMapValue @Default(values = "Search posts…")
    private String placeholder;

    @ValueMapValue @Default(intValues = 12)
    private int count;

    public String getHeading() { return heading; }
    public String getPlaceholder() { return placeholder; }
    public int getCount() { return count; }
}
```
