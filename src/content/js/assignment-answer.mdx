```js
// directory.clientlibs.js
(() => {
  'use strict';

  const SELECTORS = {
    root: '.directory',
    search: '.directory__search',
    status: '.directory__status',
    list: '.directory__list'
  };

  // pure: raw API users -> the shape we render, sorted, no mutation
  const shape = (users) =>
    users
      .map((u) => ({ name: u.name, email: u.email, company: u.company.name }))
      .sort((a, b) => a.name.localeCompare(b.name));

  const renderList = (listEl, people) => {
    const frag = document.createDocumentFragment();
    people.forEach((person) => {
      const li = document.createElement('li');
      li.className = 'directory__item';
      li.dataset.name = person.name.toLowerCase();
      li.textContent = person.name + ' — ' + person.company;
      frag.appendChild(li);
    });
    listEl.replaceChildren(frag);
  };

  const init = async (root) => {
    const search = root.querySelector(SELECTORS.search);
    const status = root.querySelector(SELECTORS.status);
    const list = root.querySelector(SELECTORS.list);
    if (!search || !status || !list) return;

    let people = [];

    // fetch + states
    status.textContent = 'Loading…';
    try {
      const res = await fetch(root.dataset.endpoint);
      if (!res.ok) throw new Error('HTTP ' + res.status);
      people = shape(await res.json());
    } catch (err) {
      status.textContent = 'Could not load the team: ' + err.message;
      return;
    }

    renderList(list, people);

    const update = () => {
      const q = search.value.trim().toLowerCase();
      const items = list.querySelectorAll(SELECTORS.list + ' > li');
      let shown = 0;
      items.forEach((li) => {
        const match = li.dataset.name.includes(q);
        li.hidden = !match;
        if (match) shown += 1;
      });
      status.textContent = 'showing ' + shown + ' of ' + people.length;
    };

    search.addEventListener('input', update);
    update();
  };

  const initAll = () => document.querySelectorAll(SELECTORS.root).forEach(init);

  document.addEventListener('DOMContentLoaded', initAll);
})();
```

## Why it's shaped this way

- **Functional structure** (lessons 03, 15): an IIFE, `SELECTORS`, `init(root)`/`initAll`, config read from `root.dataset.endpoint` — state (`people`) lives in the closure.
- **Fetch with states** (lessons 10, 14): loading first, `res.ok` checked, errors caught and shown — it never throws uncaught.
- **Pure transform** (lessons 06, 08): `shape` maps + sorts a **copy**; given the same response it always returns the same list, so it's trivial to reason about.
- **Fragment render** (lesson 11): one DOM write, `textContent` (no injection), `data-name` for filtering.
- **Live filter** (lessons 05, 06): one `input` listener, toggles `hidden`, keeps a running count.

<Callout type="do">Notice the side effects (fetch, DOM writes) sit at the edges of `init`, while `shape` stays pure — that split is what makes the logic easy to follow and change.</Callout>
