<Callout type="note">One correct solution — build the [checkpoint](checkpoint-assignment) Toggle yourself first. The habits that matter: scope to `this.element`, idempotent `init()`, and announce changes via a custom event.</Callout>

```js
// publish/components/toggle/toggle.clientlibs.js
import { register } from '@netcentric/component-loader';

class Toggle {
  constructor(element, options) {                   // the loader calls this
    this.element = element;
    this.open = Boolean(options && options.open);    // from data-nc-params
    this.init();
  }

  init() {
    if (this.element.dataset.ncReady) return;        // idempotent — safe to run twice
    this.element.dataset.ncReady = 'true';

    // DOM refs scoped to this.element
    this.button = this.element.querySelector('.toggle__button');
    this.panel = this.element.querySelector('.toggle__panel');
    this.setOpen(this.open);

    this.button.addEventListener('click', () => this.setOpen(!this.open));
  }

  setOpen(open) {
    this.open = open;
    this.panel.hidden = !open;
    this.button.setAttribute('aria-expanded', String(open));
    // announce the change — anyone can listen, no direct references
    this.element.dispatchEvent(new CustomEvent('toggle:change', {
      bubbles: true,
      detail: { open }
    }));
  }
}

register({ Toggle });
export default Toggle;
```

<Callout type="do">Two Toggles on a page work independently (scoped refs), it survives a double `init()`, and it announces every change via `toggle:change`.</Callout>
